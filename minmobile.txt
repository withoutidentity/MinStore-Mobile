1.npx create-next-app@latest // ก่อนรันต้อง cd my-app ก่อนถึงจะรันได้ เพราะว่าเราสร้างโฟลเดอร์แยก

backend 
1.npm init //enter รัวๆ
2.npm i express 
3.npm i body-parser // จัด json
4.npm i cors //ทำให้หลังบ้านติดต่อ api เราได้
5.npm install typescript tsx @types/node //เพื่อระบุไทป์ดาต้า ดีไซน์โมเดล 
6.npx tsc --init
7.npm i prisma
8.npx prisma init --datasource-pro vider mongodb
9.npx prisma generate //ทุกครั้งที่สร้าง model ใหม่
10.npm i jsonwebtoken
11.npm i dotenv
12.npx prisma db push //อัพเดทโมเดลที่เราสร้างใน schema.prisma ให้เข้า mongodb
CMD node --watch server.js
